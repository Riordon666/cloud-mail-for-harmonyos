# 05 — 导航与路由

> ArkUI 导航对标 Vue Router。组件内导航 vs 页面间路由。

## 导航方案对照

| ArkUI | Vue Router 对标 | 适用场景 |
|-------|----------------|---------|
| `Navigation` + `NavPathStack` | `<router-view>` + `router.push()` | 页面栈管理 |
| `Tabs` + `TabContent` | Tab 路由 | 底部 Tab 切换 |
| `HdsNavigation` + `HdsTabs` | 增强版 Tab 路由 | 沉浸式 Tab 框架 ✅ 当前使用 |
| `Router` (旧) | 简单页面跳转 | 不推荐（被 Navigation 替代） |

## HdsNavigation + HdsTabs（当前项目方案）

```typescript
import { hdsMaterial, HdsNavigation, HdsTabs,
         HdsTabsController, ScrollEffectType } from '@kit.UIDesignKit'

@Entry
@Component
struct Index {
  @State currentTab: number = 0
  private tabController: HdsTabsController = new HdsTabsController()

  aboutToAppear(): void {
    this.tabController.preloadItems([0, 1, 2, 3, 4])  // 预加载
  }

  build() {
    HdsNavigation() {
      HdsTabs({ controller: this.tabController, index: this.currentTab }) {
        TabContent() { InboxView() }
          .tabBar(this.TabBarItem('收件箱', 0))
        TabContent() { StarredView() }
          .tabBar(this.TabBarItem('星标', 1))
        // ... 更多 Tab
      }
      .scrollable(false)            // 禁止滑动切换
      .barOverlap(true)             // 内容穿透 Tab 栏
      .vertical(false)
      .barPosition(BarPosition.End)
      .barFloatingStyle({           // 悬浮样式
        barBottomMargin: 36,
        adaptToHandedness: true,
        systemMaterialEffect: {
          materialType: hdsMaterial.MaterialType.ADAPTIVE,
          materialLevel: hdsMaterial.MaterialLevel.ADAPTIVE
        }
      })
      .onChange((index: number) => { this.currentTab = index })
    }
    .mode(NavigationMode.Stack)
    .titleBar({
      style: {
        scrollEffectOpts: {         // 沉浸光感
          enableScrollEffect: true,
          scrollEffectType: ScrollEffectType.IMMERSIVE_GRADIENT_BLUR
        },
        systemMaterialEffect: {
          materialType: hdsMaterial.MaterialType.ADAPTIVE,
          materialLevel: hdsMaterial.MaterialLevel.ADAPTIVE
        }
      }
    })
    .hideBackButton(true)
    .ignoreLayoutSafeArea(
      [LayoutSafeAreaType.SYSTEM],
      [LayoutSafeAreaEdge.TOP, LayoutSafeAreaEdge.BOTTOM]
    )
  }

  @Builder TabBarItem(text: string, index: number) {
    Column() {
      Text(text)
        .fontSize(11)
        .fontWeight(this.currentTab === index
          ? FontWeight.Bold : FontWeight.Normal)
        .fontColor(this.currentTab === index
          ? $r('app.color.primary') : $r('app.color.text_tertiary'))
    }
    .justifyContent(FlexAlign.Center)
    .width('100%').height('100%')
  }
}
```

**关键设计参数**（来自官方 UI Design Kit 文档）：
- `MaterialType.ADAPTIVE` — 随系统自适应材质
- `MaterialType.IMMERSIVE` — 沉浸式材质（更强模糊）
- `MaterialLevel.ADAPTIVE/EXQUISITE/GENTLE/SMOOTH` — 材质级别
- `ScrollEffectType.IMMERSIVE_GRADIENT_BLUR` — 滚动渐隐毛玻璃

## Navigation + NavPathStack（页面栈导航）

用于 Tab 内部子页面跳转（如：收件箱 → 邮件详情）。

```typescript
// 主页面
@Entry
@Component
struct InboxPage {
  @Provide('navPathStack') pageStack: NavPathStack = new NavPathStack()

  build() {
    Navigation(this.pageStack) {
      List() { /* 邮件列表 */ }
    }
    .title('收件箱')
    .mode(NavigationMode.Stack)
  }
}

// 跳转到详情
this.pageStack.pushPathByName('MessagePage',
  { emailId: 123, subject: '标题' })

// 详情页
@Builder
function MessagePageBuilder(name: string, param: Object) {
  MessagePage({ emailId: param['emailId'] })
}

// 注册路由（在 aboutToAppear 或页面初始化时）
this.pageStack.setBuilder('MessagePage', wrapBuilder(MessagePageBuilder))

// 详情页中返回
@Component
struct MessagePage {
  @Consume('navPathStack') pageStack: NavPathStack

  build() {
    Column() {
      Button('返回').onClick(() => { this.pageStack.pop() })
      Text('邮件详情')
    }
  }
}
```

## 基础 Tabs 用法（备选方案）

```typescript
Tabs({ index: this.currentTab }) {
  TabContent() { InboxPage() }
    .tabBar(this.TabBuilder('收件箱', 0, 'sys.symbol.envelope'))
  TabContent() { StarredPage() }
    .tabBar(this.TabBuilder('星标', 1, 'sys.symbol.star'))
}
.barHeight(56)
.barPosition(BarPosition.End)
.barBackgroundColor(this.isDark() ? '#1A1A1ACC' : '#FFFFFFCC')
.barBackBlurStyle(BlurStyle.BACKGROUND_REGULAR)
.onChange((index: number) => { this.currentTab = index })
```

## 页面间路由（Router API，不推荐新项目使用）

```typescript
import { router } from '@kit.ArkUI'

// 跳转
router.pushUrl({ url: 'pages/MessagePage', params: { emailId: 123 } })

// 返回
router.back()

// 替换（如登录成功后跳转主页）
router.replaceUrl({ url: 'pages/Index' })

// 清空栈后跳转
router.clear()
router.pushUrl({ url: 'pages/Index' })
```

## Cloud Mail 导航架构建议

```
Index.ets (入口，HdsNavigation + HdsTabs)
├── Tab 0: InboxView
│   └── 点击邮件 → Navigation.push → MessagePage (详情)
│       ├── 回复 → Navigation.push → ComposePage
│       └── 附件 → Navigation.push → AttachmentPage
├── Tab 1: StarredView
│   └── 同 InboxView 子页面
├── Tab 2: ComposeView
├── Tab 3: AccountsView
│   └── 添加账号 → Navigation.push → AddAccountPage
└── Tab 4: MeView
    ├── 用户管理 → Navigation.push → AdminUsersPage
    ├── 全部邮件 → Navigation.push → AdminEmailsPage
    ├── 角色权限 → Navigation.push → AdminRolesPage
    ├── 数据分析 → Navigation.push → AdminAnalysisPage
    └── 注册码 → Navigation.push → AdminRegKeyPage

独立页面（不在 Tab 内）:
├── LoginPage (未登录时作为入口)
└── RegisterPage
```

**路由守卫模式**（对标原项目 router.beforeEach）：
```typescript
aboutToAppear(): void {
  const token = AppStorage.get<string>('token')
  if (!token || token === '') {
    router.replaceUrl({ url: 'pages/LoginPage' })
  }
}
```
