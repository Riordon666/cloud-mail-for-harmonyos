```typescript
import { http } from '@kit.NetworkKit'

const request = http.createHttp()

// GET 请求
const getResponse = await request.request('https://riordon.xyz/api/email/list', {
  method: http.RequestMethod.GET,
  header: {
    'Content-Type': 'application/json',
    'Authorization': 'Bearer ' + token
  },
  extraData: { accountId: 1, type: 0, size: 20 }
})

// POST 请求
const postResponse = await request.request('https://riordon.xyz/api/login', {
  method: http.RequestMethod.POST,
  header: { 'Content-Type': 'application/json' },
  extraData: JSON.stringify({ email: 'admin@riordon.xyz', password: '***' })
})

// 处理响应
if (postResponse.responseCode === 200) {
  const data = JSON.parse(postResponse.result as string)
  console.log(data)
}

// 销毁
request.destroy()
```

## HttpClient 封装（对标原项目 axios/）

```typescript
// api/HttpClient.ets
import { http } from '@kit.NetworkKit'
import Constants from '../common/Constants'

export class HttpClient {
  private static instance: HttpClient
  private baseUrl: string = Constants.API_URL  // 'https://riordon.xyz/api'

  static getInstance(): HttpClient {
    if (!HttpClient.instance) {
      HttpClient.instance = new HttpClient()
    }
    return HttpClient.instance
  }

  private getHeaders(): Record<string, string> {
    const token = AppStorage.get<string>('token') ?? ''
    return {
      'Content-Type': 'application/json',
      'Authorization': token ? `Bearer ${token}` : '',
    }
  }

  async get<T>(path: string, params?: Record<string, Object>): Promise<ApiResponse<T>> {
    const request = http.createHttp()
    try {
      const url = this.baseUrl + path + this.toQueryString(params)
      const response = await request.request(url, {
        method: http.RequestMethod.GET,
        header: this.getHeaders()
      })
      return this.handleResponse<T>(response)
    } finally {
      request.destroy()
    }
  }

  async post<T>(path: string, body?: Object): Promise<ApiResponse<T>> {
    const request = http.createHttp()
    try {
      const response = await request.request(this.baseUrl + path, {
        method: http.RequestMethod.POST,
        header: this.getHeaders(),
        extraData: JSON.stringify(body ?? {})
      })
      return this.handleResponse<T>(response)
    } finally {
      request.destroy()
    }
  }

  async delete<T>(path: string, params?: Record<string, Object>): Promise<ApiResponse<T>> {
    const request = http.createHttp()
    try {
      const url = this.baseUrl + path + this.toQueryString(params)
      const response = await request.request(url, {
        method: http.RequestMethod.DELETE,
        header: this.getHeaders()
      })
      return this.handleResponse<T>(response)
    } finally {
      request.destroy()
    }
  }

  private handleResponse<T>(response: http.HttpResponse): ApiResponse<T> {
    if (response.responseCode === 401) {
      AppStorage.setOrCreate('token', '')
      router.replaceUrl({ url: 'pages/LoginPage' })
      return { ok: false, error: '认证过期' }
    }
    const body = JSON.parse(response.result as string)
    return { ok: true, data: body.data as T, total: body.total }
  }

  private toQueryString(params?: Record<string, Object>): string {
    if (!params || Object.keys(params).length === 0) return ''
    const kv = Object.entries(params)
      .filter(([_, v]) => v !== undefined && v !== null)
      .map(([k, v]) => `${k}=${encodeURIComponent(String(v))}`)
      .join('&')
    return kv ? '?' + kv : ''
  }
}

interface ApiResponse<T> {
  ok: boolean
  data?: T
  total?: number
  error?: string
}
```

## API 模块示例（对标原项目 request/）

```typescript
// api/EmailApi.ets
import { HttpClient } from './HttpClient'

export class EmailApi {
  static async fetchList(accountId: number, type: number,
    size: number = 20, emailId?: number): Promise<EmailListData> {
    const res = await HttpClient.getInstance()
      .get<EmailData[]>('/email/list', { accountId, type, size, emailId })
    if (!res.ok) throw new Error(res.error)
    return { list: res.data ?? [], total: res.total ?? 0 }
  }

  static async sendEmail(data: ComposeData): Promise<EmailData> {
    const res = await HttpClient.getInstance()
      .post<EmailData>('/email/send', data)
    if (!res.ok) throw new Error(res.error)
    return res.data!
  }

  static async deleteEmails(emailIds: number[]): Promise<void> {
    const res = await HttpClient.getInstance()
      .delete<void>('/email/delete', { emailIds: emailIds.join(',') })
    if (!res.ok) throw new Error(res.error)
  }
}

// api/AuthApi.ets
export class AuthApi {
  static async login(email: string, password: string): Promise<string> {
    const res = await HttpClient.getInstance()
      .post<{ token: string }>('/login', { email, password })
    if (!res.ok) throw new Error(res.error)
    return res.data!.token
  }

  static async register(email: string, password: string,
    regCode?: string): Promise<string> {
    const res = await HttpClient.getInstance()
      .post<{ token: string }>('/register', { email, password, regCode })
    if (!res.ok) throw new Error(res.error)
    return res.data!.token
  }
}

// api/AccountApi.ets
export class AccountApi {
  static async fetchList(): Promise<AccountData[]> {
    const res = await HttpClient.getInstance()
      .get<AccountData[]>('/account/list')
    if (!res.ok) throw new Error(res.error)

    const res = await HttpClient.getInstance()
      .post<AccountData>('/account/add', { email, name })
    if (!res.ok) throw new Error(res.error)
    return res.data!
  }

  static async deleteAccount(accountId: number): Promise<void> {
    const res = await HttpClient.getInstance()
      .delete<void>('/account/delete', { accountId })
    if (!res.ok) throw new Error(res.error)
  }
}
```

## 附件上传/下载

```typescript
// 下载附件（带进度、缓存）
async function downloadWithCache(
  url: string, filename: string): Promise<string> {
  // 先查缓存
  const cachePath = getContext().cacheDir + '/att/' + filename
  if (fileIo.accessSync(cachePath)) {
    return cachePath   // 命中缓存
  }

  const request = http.createHttp()
  const response = await request.request(url)
  const file = fileIo.openSync(cachePath, fileIo.OpenMode.CREATE
    | fileIo.OpenMode.WRITE_ONLY)
  fileIo.writeSync(file.fd, response.result as ArrayBuffer)
  fileIo.closeSync(file.fd)
  request.destroy()
  return cachePath
}

// 上传附件（multipart）
async function uploadFile(
  filePath: string, uploadUrl: string): Promise<string> {
  const request = http.createHttp()
  const response = await request.request(uploadUrl, {
    method: http.RequestMethod.POST,
    header: {
      'Content-Type': 'multipart/form-data',
      'Authorization': 'Bearer ' + AppStorage.get<string>('token')
    },
    extraData: filePath
  })
  const body = JSON.parse(response.result as string)
  request.destroy()
  return body.data.key  // 返回文件 key
}
```

## 错误处理

```typescript
// 统一错误拦截（对标原项目 axios interceptors）
async function requestWithRetry<T>(
  fn: () => Promise<T>, maxRetries: number = 3): Promise<T> {
  for (let i = 0; i < maxRetries; i++) {
    try {
      return await fn()
    } catch (err) {
      if (i === maxRetries - 1) throw err
      if (err.message?.includes('401')) throw err   // 认证失败不重试
      await sleep(1000 * (i + 1))   // 指数退避
    }
  }
  throw new Error('Max retries exceeded')
}

function sleep(ms: number): Promise<void> {
  return new Promise(resolve => setTimeout(resolve, ms))
}
```
