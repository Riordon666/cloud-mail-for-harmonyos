# 07 — 文件管理

> Core File Kit 对标原项目 R2 对象存储

## 应用沙箱目录

```
/data/storage/el2/base/haps/<bundleName>/
├── files/              # 持久文件（备份恢复）
├── cache/              # 缓存文件（系统可清理）
├── preferences/        # Preferences 数据
├── temp/               # 临时文件（应用退出即删）
├── database/           # RelationalStore 数据库
└── distributedfiles/   # 跨设备共享文件
```

**获取路径**：
```typescript
import { UIAbility } from '@kit.AbilityKit'

// 在 Ability 中
const context = this.context
const filesDir = context.filesDir        // /data/.../files
const cacheDir = context.cacheDir        // /data/.../cache
const tempDir = context.tempDir          // /data/.../temp
const dbDir = context.databaseDir        // /data/.../database
const distDir = context.distributedFilesDir
```

**加密级别**：
- EL1: 开机即可用（闹钟等）
- EL2: 解锁后可用（默认，大多数应用）
- EL3: 解锁后可用（更高安全）
- EL4: 解锁后可用（最高安全）
- EL5: 解锁后可用（特殊保密）

## 文件读写操作

```typescript
import { fileIo } from '@kit.CoreFileKit'

// 写入文件
const filePath = context.filesDir + '/draft_email.txt'
const file = fileIo.openSync(filePath, fileIo.OpenMode.CREATE
  | fileIo.OpenMode.WRITE_ONLY)
fileIo.writeSync(file.fd, '邮件草稿内容...')
fileIo.closeSync(file.fd)

// 读取文件
const readFile = fileIo.openSync(filePath, fileIo.OpenMode.READ_ONLY)
const buf = new ArrayBuffer(1024)
const readLen = fileIo.readSync(readFile.fd, buf)
const content = String.fromCharCode.apply(
  null, new Uint8Array(buf, 0, readLen))
fileIo.closeSync(readFile.fd)

// 检查文件是否存在
const exists = fileIo.accessSync(filePath)

// 删除文件
fileIo.unlinkSync(filePath)

// 获取文件信息
const stat = fileIo.statSync(filePath)
console.log(`size: ${stat.size}, mtime: ${stat.mtime}`)
```

## 附件处理方案

### 1. 附件下载与缓存

```typescript
import { http } from '@kit.NetworkKit'
import { fileIo } from '@kit.CoreFileKit'

async function downloadAttachment(
  url: string, filename: string, context: Context): Promise<string> {
  // 下载到缓存目录
  const cachePath = context.cacheDir + '/attachments/'
  fileIo.mkdirSync(cachePath, true)  // 确保目录存在

  const filePath = cachePath + filename
  const request = http.createHttp()
  const response = await request.request(url)
  const file = fileIo.openSync(filePath, fileIo.OpenMode.CREATE
    | fileIo.OpenMode.WRITE_ONLY)
  fileIo.writeSync(file.fd, response.result as ArrayBuffer)
  fileIo.closeSync(file.fd)
  request.destroy()
  return filePath
}
```

### 2. 附件上传

```typescript
async function uploadAttachment(
  filePath: string, uploadUrl: string): Promise<boolean> {
  const request = http.createHttp()
  const response = await request.request(uploadUrl, {
    method: http.RequestMethod.POST,
    header: { 'Content-Type': 'multipart/form-data' },
    extraData: filePath    // 直接传文件路径
  })
  request.destroy()
  return response.responseCode === 200
}
```

### 3. 打开/预览附件

```typescript
import { Want } from '@kit.AbilityKit'

// 调用系统预览
function previewFile(context: Context, filePath: string) {
  const want: Want = {
    action: 'ohos.want.action.viewData',
    uri: 'file://' + filePath,
    type: getMimeType(filePath)  // 'application/pdf', 'image/png' 等
  }
  context.startAbility(want)
}

function getMimeType(filename: string): string {
  const ext = filename.split('.').pop()?.toLowerCase()
  const mimeMap: Record<string, string> = {
    'pdf':  'application/pdf',
    'png':  'image/png',
    'jpg':  'image/jpeg',
    'doc':  'application/msword',
    'docx': 'application/vnd.openxmlformats-officedocument.wordprocessingml.document',
    'xls':  'application/vnd.ms-excel',
    'ppt':  'application/vnd.ms-powerpoint',
    'zip':  'application/zip',
    'txt':  'text/plain',
  }
  return mimeMap[ext ?? ''] ?? 'application/octet-stream'
}
```

## 文件选择器（用户选择附件）

```typescript
import { picker } from '@kit.CoreFileKit'

// 选择单个文件
async function pickAttachment(): Promise<string[]> {
  const documentPicker = new picker.DocumentViewPicker()
  const options = new picker.DocumentSelectOptions()
  options.maxSelectNumber = 5
  options.fileSuffixFilters = ['pdf', 'doc', 'docx', 'xls', 'ppt',
                                'jpg', 'png', 'zip', 'txt']

  try {
    const uris = await documentPicker.select(options)
    return uris   // ['file://docs/storage/.../xxx.pdf', ...]
  } catch (err) {
    console.error('选择文件失败:', err)
    return []
  }
}

// 保存文件到用户目录
async function saveToUserDir(sourcePath: string, filename: string) {
  const filePicker = new picker.DocumentViewPicker()
  const saveOptions = new picker.DocumentSaveOptions()
  saveOptions.newFileNames = [filename]

  const targetUri = await filePicker.save(saveOptions)
  // 复制文件到用户选择的位置
  fileIo.copyFileSync(sourcePath, targetUri)
}
```

## 存储决策

| 数据类型 | 目录 | 原因 |
|---------|------|------|
| 下载的附件 | `cacheDir/attachments/` | 可被系统清理 |
| 保存的附件 | `filesDir/saved/` | 持久保存 |
| 邮件草稿中的图片 | `cacheDir/images/` | 临时内嵌图片 |
| 用户头像缓存 | `cacheDir/avatars/` | 可重新下载 |
| 导出数据 | `filesDir/exports/` | 用户主动操作 |
| 日志文件 | `tempDir/` | 临时调试 |
